				limit = {
					NOT = { has_religion = catholic }
				}
				set_character_religion = catholic
			}
		}

		ordered_character = {
			limit = {
				loyalty >= 60
				NOT = { has_religion = catholic }
				is_ruler = no
				from_ruler_family = no
			}
			max = 15
			order_by = prominence
			set_character_religion = catholic
		}

            ordered_pops_in_province = {
					limit = {
						NOT = { pop_religion = catholic }
					}
					order_by = pop_hapiness
					max = {
						value = total_population
						divide = 2
					}

					set_pop_religion = catholic
				}
        }
