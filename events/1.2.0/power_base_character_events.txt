# CHAIN 1 - Massive Treasury and not afraid to spend it
pbc_event.1 = {
	type = country_event
	title = "pbc_event.1.t"
	desc = "pbc_event.1.desc"
	picture = council_country

	left_portrait = current_ruler
	right_portrait = scope:power_character

	trigger = {
		any_character = {
			loyalty > 33
			power_base >= 10
			is_ruler = no
			prisoner = no
			employer = root
			trigger_if = {
				limit = {
					root = {
						is_monarchy = yes
					}
				}
				is_primary_heir = no
			}
			trigger_if = {
				limit = {
					root = {
						has_co_ruler_government = yes
					}
				}
				is_co_ruler = no
			}
			NOT = { is_friend = root.current_ruler }
		}
		NOT = { has_variable = pbc_event_1_cooldown }
	}

	immediate = {
		set_variable = {
			name = pbc_event_1_cooldown
			days = 10950
		}
		random_character = {
			limit = {
				loyalty > 33
				power_base >= 10
				is_ruler = no
				prisoner = no
				employer = root
				trigger_if = {
					limit = {
						root = {
							is_monarchy = yes
						}
					}
					is_primary_heir = no
				}
				trigger_if = {
					limit = {
						root = {
							has_co_ruler_government = yes
						}
					}
					is_co_ruler = no
				}
				NOT = { is_friend = root.current_ruler }
			}
			save_scope_as = power_character
		}
		scope:power_character = {
			if = {
				limit = {
					popularity < 100
				}
				add_popularity = popularity_medium
			}
			add_corruption = corruption_medium
			add_gold = 	{
				value = root.has_monthly_income
				multiply = 5
			}
		}
	}

	option = {
		name = "pbc_event.1.a"
		if = {
			limit = {
				is_monarchy = yes
			}
			add_legitimacy = subtract_legitimacy_small
		}
		current_ruler = {
			add_popularity = subtract_popularity_small
		}
		scope:power_character = {
			add_loyalty = loyalty_growing_coffers
		}
		hidden_effect = {
			random_list = {
				5 = { # Second insult
					trigger_event = {
						id = pbc_event.2
						days = { 200 300 }
					}
				} 
				10 = {} # Nothing Happens
			}
		}
	}

	option = {
		name = "pbc_event.1.b"
		add_tyranny = tyranny_medium
		add_treasury = root.has_monthly_income
		scope:power_character = {
			add_loyalty = loyalty_appropriated_wealth
			add_gold = {
				subtract = root.has_monthly_income
			}
		}
	}

	option = {
		name = "pbc_event.1.c"
		highlight = yes
		trigger = {
			current_ruler = {
				has_trait = arbitrary
			}
		}
		custom_tooltip = "arbitrary_tooltip"
		add_tyranny = tyranny_large
		imprison = {
			target = scope:power_character
		}
		scope:power_character = {
			add_loyalty = loyalty_unjustly_imprisoned
		}
	}

	option = {
		name = "pbc_event.1.d"
		highlight = yes
		trigger = {
			current_ruler = {
				has_trait = corrupt
			}
		}
		custom_tooltip = "corrupt_tooltip"
		add_treasury = {
			subtract = root.has_monthly_income
			multiply = 2
		}
		current_ruler = {
			add_corruption = corruption_small
		}
		scope:power_character = {
			add_corruption = corruption_small
			add_loyalty = loyalty_gift_bribe_1
			add_gold = {
				value = root.has_monthly_income
				multiply = 2
			}
		}
	}
}

pbc_event.8 = {
	type = country_event
	title = "pbc_event.8.t"
	desc = "pbc_event.8.desc"
	picture = council_country

	left_portrait = current_ruler
	right_portrait = scope:power_character
	right_portrait = scope:faction_character_1
	right_portrait = scope:faction_character_2

	trigger = {
		NOT = { has_variable = pbc_event_8_cooldown }
		is_republic = yes
		has_co_ruler_government = yes
		any_character = {
			loyalty > 33
			NOT = { is_same_party_as = root.current_ruler }
			power_base >= 5
			is_ruler = no
			is_co_ruler = yes
			NOT = { is_friend = root.current_ruler }
		}
	}

	immediate = {
		set_variable = {
			name = pbc_event_8_cooldown
			days = 10950
		}
		random_character = {
			limit = {
				loyalty > 33
				NOT = { is_same_party_as = root.current_ruler }
				power_base >= 5
				is_ruler = no
				is_co_ruler = yes
				NOT = { is_friend = root.current_ruler }
			}
			save_scope_as = power_character
		}
		if = {
			limit = {
				any_character = {
					is_same_party_as = scope:power_character
					is_party_leader = no
					NOT = { this = scope:power_character }
					trigger_if = {
						limit = {
							employer = {
								gender_equality = no
							}
						}
						is_female = no
					}
				}
			}
			random_character = {
				limit = {
					is_same_party_as = scope:power_character
					is_party_leader = no
					NOT = { this = scope:power_character }
					trigger_if = {
						limit = {
							employer = {
								gender_equality = no
							}
						}
						is_female = no
					}
				}
				weight = {
					modifier = {
						factor = 3
						power_base >= 2
					}
					modifier = {
						factor = 3
						power_base >= 5
					}
				}
				save_scope_as = faction_character_1
