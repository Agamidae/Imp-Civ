namespace = roman_conquest

roman_conquest.1 = {
	type = province_event
	title = "roman_conquest.1.t"
	desc = "roman_conquest.1.desc"
	picture = romans_marching

	left_portrait = owner.current_ruler
	
	goto_location = root

	trigger = {
		owner = {
			tag = ROM
			primary_culture = roman
		}
		total_population > 20
		NOT = { dominant_province_culture = roman }
		OR = {
			province_dominant_culture_group = latin
			province_dominant_culture_group = hellenic
		}
		owner = { NOT = { has_variable = roman_conquest_1_cooldown } }
	}

	immediate = {
		owner = {
			set_variable = {
				name = roman_conquest_1_cooldown
				days = 365
			}
		}
	}

	option = {
		name = "roman_conquest.1.a"
		owner = {
			add_tyranny = tyranny_small
		}
		custom_tooltip = "roman_conquest.1.tt"
		add_province_modifier = {
			name = province_improvement_roman_conquest_loyalty
			duration = 3650
		}
		hidden_effect = {
			ordered_pops_in_province = {
				limit = {
					OR = {
						pop_culture_group = latin
						pop_culture_group = hellenic
					}
					NOR = { 
						pop_culture = roman 
						pop_type = slaves
					}
				}
				order_by = pop_hapiness
				max = {
					value = total_population
					divide = 5
				}
				set_pop_culture = roman
			}
		}
	}

	option = {
		name = "roman_conquest.1.b"
		custom_tooltip = "no_effect_tooltip"
	}
}

roman_conquest.2 = {							# 1 - End Etruscans
    type = country_event
    title = roman_conquest.2.t
    desc = {
		desc = roman_conquest.2.desc
		desc = roman_conquest.colonia_desc
	}
    picture = romans_marching
    
	fire_only_once = yes
	
    trigger = {
		has_land = yes
    }

    immediate = {
		if = {
			limit = {
				any_subject = {
					OR = {
						tag = ETR
						primary_culture = etruscan

							OR = {
								is_in_area = north_etruria_area
								is_in_area = south_etruria_area

						}
					}
				}
			}
			every_subject = {
				limit = {
					OR = {
						tag = ETR

							OR = {
								is_in_area = north_etruria_area
								is_in_area = south_etruria_area
							}
						}
					}
				}
				add_opinion = { target = c:ROM modifier = rm_new_conquest_subject_opmod }

		}

    option = {							# Volci Colonia 
        name = roman_conquest.2.a

			owns = 22
		}

			owns = 116
		}

			owns = 114
		}

			roman_mission_colonia_effect = yes

		}

			roman_mission_colonia_effect = yes

roman_conquest.7 = {							# 1 - Galvanize Gauls
    type = country_event
    title = roman_conquest.7.t
   desc = {
		desc = roman_conquest.7.desc
		desc = roman_conquest.colonia_desc
	}
    picture = romans_marching
    
	fire_only_once = yes

    trigger = {
		has_land = yes
    }

    immediate = {
		if = {
			limit = {
				any_owned_province = {
					is_in_area = ariminum_area
					province_dominant_culture_group = gallic
				}
			}
			if = {
				limit = {
					any_neighbour_country = {
						has_land = yes
						country_culture_group = gallic
					}
				}
				ordered_neighbour_country = {
					limit = {
						has_land = yes
						country_culture_group = gallic
						NOT = { primary_culture = hebrew }
					}
					order_by = country_population
					save_scope_as = rm_gallic_exodus_country
				}
			}
			else_if = {
				limit = {
					any_country = {
						has_land = yes
						country_culture_group = gallic
						in_diplomatic_range = c:ROM
					}
				}
				random_country = {
					limit = {
						has_land = yes
						country_culture_group = gallic
						in_diplomatic_range = c:ROM
					}
					weight = {
						modifier = {
							factor = 100
							primary_culture = lepontic
						}
					}
					save_scope_as = rm_gallic_exodus_country
				}
			}
			else = {
				random_country = {
					limit = {
						has_land = yes
						in_diplomatic_range = c:ROM
					}
					save_scope_as = rm_gallic_exodus_country
				}
			}
		}
		rm_gallic_exodus_effect = yes
    }

    option = {      
        name = roman_conquest.7.a
		trigger = {
			owns = 121
		}
		p:121 = {
			roman_mission_colonia_effect = yes
		}
    }
	
    option = {

		}

			roman_mission_colonia_effect = yes
		}
    }
	
	option = {      
        name = roman_conquest.7.c
		trigger = {
			owns = 142
		}
		p:142 = {
			roman_mission_colonia_effect = yes
